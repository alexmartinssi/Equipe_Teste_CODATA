    Input Text     xpath=//input[@id='formAdicionaUsuario:adicionaTelefone']  8332312526
    Select From List 	xpath=//select[contains(@id,'formAdicionaUsuario:adicionaPerfil')]  Usuário
    Select From List 	xpath=//select[contains(@id,'formAdicionaUsuario:adicionaEntidade')]  Secretaria de Estado da Mulher e da Diversidade Humana
    Click Element   id=formAdicionaUsuario:adicionaAtivo
    Click Element    xpath=//button[contains(@id,'formAdicionaUsuario')]/span[text()='Confirmar']
